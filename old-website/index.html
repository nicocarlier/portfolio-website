Tubify

Overview
A simple single page transcription tool which allows users to transform a Youtube video to text through the Youtube API and GPT3.5 Turbo API.

Code snippet
The Youtube transcription API was only compatiple with python, so a custom fetching route served as middleware between the python and javascript. 

								
app.get('/extract/:videoId', (req, res) => {
	const spawn = require('child_process').spawn;
	const videoId = req.params.videoId;
	const pythonProcess = spawn('python3', ['./transcripter.py', videoId]);
	let outputData = '';

	pythonProcess.stdout.on('data', function(data) {
		outputData += data.toString();
	});
	
	pythonProcess.on('close', (code) => {
		if(code !== 0) {
		console.error("Python process exited with code " + code);
		} else {
		res.send(outputData);
		}
	});
	
	pythonProcess.stderr.on('data', (data) => {
		console.error("Error:", data.toString());
	});
});
